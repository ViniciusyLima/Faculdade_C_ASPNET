        {
            this.Nome = nome;

        }

        {
            get => nome;
            set => nome = value;
        }
        public float Salario
        {
            get => salario;
            set => salario = value;
        }

        public string toString()
        {
           return nome + ", sal치rio R$" + Salario;
        }
    }
}
